def scan_section(section, param={})
  index = 0

  hits  = [ ]

  while index < section.size && (index = section.index(regex, index)) != nil

    idx = index
    buf = ''
    mat = nil

    while (! (mat = buf.match(regex)))
      buf << section.read(idx, 1)
      idx += 1
    end

    rva = section.offset_to_rva(index)

    hits << [ rva, buf.unpack("H*") ]
    index += buf.length
  end

  return hits
end
